Append more products dynamically using JavaScript. This is how the loadmore button works.

function openModal(d){
  modal.style.display='flex';
  imgEl.src=d.image;
  nameEl.textContent=d.name;
  descEl.textContent=d.description;
  priceEl.textContent=d.price;
  basePrice=+d.price;

    btn.textContent=s;
    if(i===0){btn.classList.add('active');selectedSize=s;}

  const saved=JSON.parse(localStorage.getItem('customer')||'{}');
  if(saved.name)custName.value=saved.name;
